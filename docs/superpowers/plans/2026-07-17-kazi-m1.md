**Goal:** Every HTTP service reachable at `https://<stack>.localhost` (green lock after one `kazi trust`), stable opt-in host ports for TCP services via `kazi expose`, and `kazi urls` as the single answer to "where is everything?"

**Architecture:** A new `internal/proxy` package holds all pure logic (classification, Caddyfile rendering, override rendering, port allocation) plus the proxy-stack manager. The engine wires a `syncProxy` (render â†’ diff â†’ validate â†’ reload, warning-only) into every state-changing verb. Caddy runs as a kazi-generated compose stack (`kazi-proxy`, `system: true` manifest) on a shared external `kazi` network; routed services get network-scoped aliases `<service>.<stack>` to avoid name collisions.

**Tech Stack:** Existing deps only (cobra, yaml.v3). Caddy `caddy:2-alpine` as a runtime container, never a Go dependency.

## Global Constraints

- All logic in `internal/engine` + `internal/proxy`; cobra handlers stay thin. No capability may exist only in a skin.
- Schema stays `kazi.dev/v1alpha1`; M1 manifest fields (`spec.proxy`, `spec.expose`, `spec.system`) are additive â€” M0 manifests remain valid.
- kazi never publishes a host port the compose file doesn't declare, except via explicit `spec.expose` entries.
- Proxy/reload failures NEVER block stack verbs â€” warnings on stderr (`kazi: warning: â€¦`), never errors.

```go
package compose

import (
	"encoding/json"
	"fmt"
	"sort"
	"strconv"
)

type ServiceInfo struct {
	Name      string
	Ports     []int
	Published map[int]int
	Networks  []string
}

type rawConfig struct {
	Services map[string]rawService `json:"services"`
}

type rawService struct {
	Ports []struct {
		Target    int    `json:"target"`
		Published string `json:"published"`
	} `json:"ports"`
	Expose   []any          `json:"expose"`
	Networks map[string]any `json:"networks"`
}

// ParseConfig extracts the per-service port/network view from
// `<runtime> compose config --format json` output.
func ParseConfig(jsonBytes []byte) ([]ServiceInfo, error) {
	var raw rawConfig
	if err := json.Unmarshal(jsonBytes, &raw); err != nil {
		return nil, fmt.Errorf("parsing compose config JSON: %w", err)
	}
	var out []ServiceInfo
	for name, rs := range raw.Services {
		si := ServiceInfo{Name: name, Published: map[int]int{}}
		seen := map[int]bool{}
		for _, p := range rs.Ports {
			if p.Target == 0 {
				continue
			}
			if !seen[p.Target] {
				seen[p.Target] = true
				si.Ports = append(si.Ports, p.Target)
			}
			if hp, err := strconv.Atoi(p.Published); err == nil && hp > 0 {
				si.Published[p.Target] = hp
			}
		}
		for _, e := range rs.Expose {
			var port int
			switch v := e.(type) {
			case float64:
				port = int(v)
			case string:
				port, _ = strconv.Atoi(v)
			}
			if port > 0 && !seen[port] {
				seen[port] = true
				si.Ports = append(si.Ports, port)
			}
		}
		sort.Ints(si.Ports)
		for n := range rs.Networks {
			si.Networks = append(si.Networks, n)
		}
		sort.Strings(si.Networks)
		out = append(out, si)
	}
	sort.Slice(out, func(i, j int) bool { return out[i].Name < out[j].Name })
	return out, nil
}
```

Classification rules (spec Â§Service classification): a service with no ports is None. If any port is on the HTTP list â†’ HTTP, port = the lowest such port. Else if ALL ports are on the TCP list â†’ TCP, port = lowest. Else if exactly one port (necessarily not on the TCP list here) â†’ HTTP on it. Else None. `decl.Enabled == false` â‡’ empty plan (nothing routable). Primary = `decl.Service` if set (port = `decl.HTTPPort`, else its classified HTTP port, else its single port); else the stack's single HTTP service; multiple HTTP + no decl â‡’ no bare hostname, `NeedsDecl = true`. Primary gets `<stack>.localhost`; every other HTTP service gets `<service>.<stack>.localhost`. ALL HTTP services (primary included) are Routable with alias `<service>.<stack>`.

```go
func Classify(svc compose.ServiceInfo, httpPorts, tcpPorts []int) (Class, int) {
	if len(svc.Ports) == 0 {
		return ClassNone, 0
	}
	httpSet, tcpSet := toSet(httpPorts), toSet(tcpPorts)
	for _, p := range svc.Ports { // Ports sorted asc â‡’ first hit is lowest
		if httpSet[p] {
			return ClassHTTP, p
		}
	}
	allTCP := true
	for _, p := range svc.Ports {
		if !tcpSet[p] {
			allTCP = false
			break
		}
	}
	if allTCP {
		return ClassTCP, svc.Ports[0]
	}
	if len(svc.Ports) == 1 {
		return ClassHTTP, svc.Ports[0]
	}
	return ClassNone, 0
}

func toSet(ps []int) map[int]bool {
	m := make(map[int]bool, len(ps))
	for _, p := range ps {
		m[p] = true
	}
	return m
}

func BuildPlan(stack string, decl *store.ProxySpec, svcs []compose.ServiceInfo, httpPorts, tcpPorts []int) Plan {
	plan := Plan{Routable: map[string]bool{}}
	if decl != nil && decl.Enabled != nil && !*decl.Enabled {
		return plan
	}
	httpPort := map[string]int{}
	var httpSvcs []string
	for _, s := range svcs {
		class, port := Classify(s, httpPorts, tcpPorts)
		switch class {
		case ClassHTTP:
			httpSvcs = append(httpSvcs, s.Name)
			httpPort[s.Name] = port
		case ClassTCP:
			plan.TCP = append(plan.TCP, TCPService{Service: s.Name, Port: port})
		}
	}
	sort.Strings(httpSvcs)
	sort.Slice(plan.TCP, func(i, j int) bool { return plan.TCP[i].Service < plan.TCP[j].Service })

	switch {
	case decl != nil && decl.Service != "":
		plan.Primary = decl.Service
		if decl.HTTPPort != 0 {
			httpPort[decl.Service] = decl.HTTPPort
		}
		if _, ok := httpPort[decl.Service]; !ok { // declared but unclassifiable: trust the declaration only with a port
			plan.Primary = ""
		} else if !contains(httpSvcs, decl.Service) {
			httpSvcs = append(httpSvcs, decl.Service)
			sort.Strings(httpSvcs)
		}
	case len(httpSvcs) == 1:
		plan.Primary = httpSvcs[0]
	case len(httpSvcs) > 1:
		plan.NeedsDecl = true
	}

	for _, name := range httpSvcs {
		plan.Routable[name] = true
		host := name + "." + stack + ".localhost"
		if name == plan.Primary {
			host = stack + ".localhost"
		}
		plan.Routes = append(plan.Routes, Route{
			Stack: stack, Service: name, Hostname: host,
			Alias: name + "." + stack, Port: httpPort[name],
		})
	}
	sort.Slice(plan.Routes, func(i, j int) bool { return plan.Routes[i].Hostname < plan.Routes[j].Hostname })
	return plan
}
```
